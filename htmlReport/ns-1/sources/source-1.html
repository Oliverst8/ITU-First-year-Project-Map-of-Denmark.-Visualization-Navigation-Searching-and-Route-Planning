 import javafx.stage.Stage;
 
 public class App extends Application {
 
     public static void main(String[] args) {
         launch(args);
     }
 
     @Override
     public void start(Stage primaryStage) throws Exception {
 
         if (!new File("chunkData/config").exists()) {

     }
